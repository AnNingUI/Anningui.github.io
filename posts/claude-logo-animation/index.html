Canvas 2D 复刻 Claude Logo 动画

背景

这是对 Anthropic 官网 When AI builds itself 文章中 SparkCellularHero 中的动画的约 80% 复刻。核心动画机制已使用 Canvas 2D 完整实现：BFS 驱动的元胞生长、Logo 形状与圆角矩形之间的形态变换、背景网格层、旋转的 Agent 火花，以及由三态状态机（growing → holding → zooming）驱动的无缝循环。

部分视觉打磨细节和性能优化尚未实现，但核心动画逻辑已完整可用。

核心算法：BFS 元胞生长

动画使用**广度优先搜索（BFS）**模拟从中心向外的有机元胞扩散：

function stepSimulation(now: number) {
  if (state !== 'growing') return;

  const occupied = new Set(agents.map(a => a.idx));

  for (const agent of agents) {
    const neighbors = adjacencyMap.get(agent.idx) || [];
    const validNeighbors = neighbors.filter(
      n => !activeCells.has(n) && !occupied.has(n)
    );

    if (validNeighbors.length > 0) {
      const targetIdx = validNeighbors[
        Math.floor(Math.random() * validNeighbors.length)
      ];
      occupied.add(targetIdx);

      activeCells.add(agent.idx);
      cellBirthTimes.set(agent.idx, now);

      agent.prevIdx = agent.idx;
      agent.idx = targetIdx;
      agent.progress = 0;
    } else {
      activeCells.add(agent.idx);
      cellBirthTimes.set(agent.idx, now);
      agent.dead = true;
    }
  }

  agents = agents.filter(a => !a.dead);

  // 边缘智能补充新 agent（最多 5 个）
  const maxAgents = 5;
  if (agents.length < maxAgents && activeCells.size < validGridPoints.length) {
    const borderCells: number[] = [];
    for (const idx of activeCells) {
      const neighbors = adjacencyMap.get(idx) || [];
      const hasEmptyNeighbor = neighbors.some(
        n => !activeCells.has(n) && !occupied.has(n)
      );
      if (hasEmptyNeighbor) borderCells.push(idx);
    }

    if (borderCells.length > 0) {
      borderCells.sort(() => Math.random() - 0.5);
      const spawnCount = Math.min(maxAgents - agents.length, borderCells.length);
      for (let i = 0; i < spawnCount; i++) {
        const spawnIdx = borderCells[i];
        const neighbors = adjacencyMap.get(spawnIdx) || [];
        const emptyNeighbors = neighbors.filter(
          n => !activeCells.has(n) && !occupied.has(n)
        );
        if (emptyNeighbors.length > 0) {
          const targetIdx = emptyNeighbors[
            Math.floor(Math.random() * emptyNeighbors.length)
          ];
          occupied.add(targetIdx);
          agents.push({ idx: targetIdx, prevIdx: spawnIdx, progress: 0 });
        }
      }
    }
  }

  // 检查是否填满
  if (activeCells.size === validGridPoints.length && agents.length === 0) {
    state = 'holding';
    stateChangeTime = now;
  }
}

核心设计：

  • occupied 集合防止多个 agent 选择同一目标
  • 随机邻居选择产生自然的扩散模式
  • 边缘智能补充新 agent（最多 5 个），加速填充

多层渲染

function draw(now: number) {
  if (!canvasRef) return;
  const ctx = canvasRef.getContext('2d')!;

  ctx.clearRect(0, 0, width, height);

  const cx = width / 2;
  const cy = height / 2 + 20;

  let cameraScale = 1.0;
  let currentGridOpacity = 1.0;
  let nextGridOpacity = 0.0;
  let zoomProgress = 0.0;

  // State machine
  if (state === 'growing') {
    const dt = now - lastSimTime;
    for (const agent of agents) {
      agent.progress = Math.min(1, agent.progress + dt / STEP_DURATION);
    }
    if (dt >= STEP_DURATION) {
      stepSimulation(now);
      lastSimTime = now;
    }
  } else if (state === 'holding') {
    if (now - stateChangeTime > HOLD_DURATION) {
      state = 'zooming';
      stateChangeTime = now;
    }
  } else if (state === 'zooming') {
    const elapsed = now - stateChangeTime;
    zoomProgress = Math.min(1, elapsed / ZOOM_DURATION);
    const t = easeInOutCubic(zoomProgress);

    const targetScale = cellSize / (GRID_SIZE * gridSpacing);
    cameraScale = 1.0 + (targetScale - 1.0) * t;

    currentGridOpacity = Math.max(0, 1.0 - t * 1.5);
    nextGridOpacity = Math.max(0, (t - 0.4) / 0.6);

    if (zoomProgress >= 1.0) {
      startCycle(now);
      requestAnimationFrame(draw);
      return;
    }
  }

  // 1. 未来景深网格（缩放时淡入）
  if (state === 'zooming') {
    ctx.save();
    ctx.globalAlpha = nextGridOpacity;
    for (let gy = 0; gy < GRID_SIZE; gy++) {
      for (let gx = 0; gx < GRID_SIZE; gx++) {
        const x = cx + (gx - 9) * gridSpacing;
        const y = cy + (gy - 9) * gridSpacing;
        ctx.fillStyle = GRID_DOT_COLOR;
        ctx.beginPath();
        ctx.arc(x, y, 1.2, 0, Math.PI * 2);
        ctx.fill();
      }
    }
    ctx.restore();
  }

  // 2. 当前景深网格（随相机缩放淡出）
  ctx.save();
  ctx.globalAlpha = currentGridOpacity;
  for (let gy = 0; gy < GRID_SIZE; gy++) {
    for (let gx = 0; gx < GRID_SIZE; gx++) {
      const x = cx + (gx - 9) * gridSpacing * cameraScale;
      const y = cy + (gy - 9) * gridSpacing * cameraScale;
      ctx.fillStyle = GRID_DOT_COLOR;
      ctx.beginPath();
      ctx.arc(x, y, 1.2 * cameraScale, 0, Math.PI * 2);
      ctx.fill();
    }
  }
  ctx.restore();

  // 3. 元胞层（形态变换）
  const morphThreshold = 0.7 + zoomProgress * 0.3;

  for (const cell of validGridPoints) {
    const gx = cell.gx;
    const gy = cell.gy;

    const x = cx + (gx - 9) * gridSpacing * cameraScale - (cellSize * cameraScale) / 2;
    const y = cy + (gy - 9) * gridSpacing * cameraScale - (cellSize * cameraScale) / 2;

    const hasCell = activeCells.has(cell.idx);
    const birthTime = cellBirthTimes.get(cell.idx);
    let scale = 0;

    if (hasCell && birthTime) {
      const t = Math.min(1, (now - birthTime) / 450);
      scale = easeOutCubic(t);
    }

    if (scale > 0) {
      const color = cellColors.get(cell.idx) || PALETTE[0];

      if (scale < morphThreshold) {
        const logoScale = Math.min(1, scale / morphThreshold);
        drawClaudeLogoShape(ctx, x, y, cellSize * cameraScale, color, logoScale);
      } else {
        drawRoundedRect(ctx, x, y, cellSize * cameraScale, cellRadius * cameraScale, color, scale);
      }
    }
  }

  // 4. Agent 层（旋转的白色 Logo 火花）
  if (state === 'growing') {
    for (const agent of agents) {
      const currX = agent.idx % GRID_SIZE;
      const currY = Math.floor(agent.idx / GRID_SIZE);
      const prevX = agent.prevIdx % GRID_SIZE;
      const prevY = Math.floor(agent.prevIdx / GRID_SIZE);

      const gx = prevX + (currX - prevX) * agent.progress;
      const gy = prevY + (currY - prevY) * agent.progress;

      const x = cx + (gx - 9) * gridSpacing;
      const y = cy + (gy - 9) * gridSpacing;

      ctx.save();
      ctx.shadowBlur = 15;
      ctx.shadowColor = AGENT_COLOR;

      const rotation = now * 0.004;
      ctx.translate(x, y);
      ctx.rotate(rotation);

      const size = cellSize * 1.1;
      ctx.scale(size / 125, size / 125);
      ctx.translate(-62.5, -62.5);
      ctx.fillStyle = '#FFFFFF';
      ctx.beginPath();
      ctx.fill(claudePath2D);
      ctx.restore();
    }
  }

  requestAnimationFrame(draw);
}

状态机与无缝循环

type State = 'growing' | 'holding' | 'zooming';

function updateStateMachine(now: number) {
  switch (state) {
    case 'growing':
      // BFS 每 150ms 步进一次
      const dt = now - lastSimTime;
      for (const agent of agents) {
        agent.progress = Math.min(1, agent.progress + dt / STEP_DURATION);
      }
      if (dt >= STEP_DURATION) {
        stepSimulation(now);
        lastSimTime = now;
      }
      break;

    case 'holding':
      // 填满后静止 1 秒
      if (now - stateChangeTime > HOLD_DURATION) {
        state = 'zooming';
        stateChangeTime = now;
      }
      break;

    case 'zooming':
      // 缩放 1.6 秒
      const elapsed = now - stateChangeTime;
      const zoomProgress = Math.min(1, elapsed / ZOOM_DURATION);
      const t = easeInOutCubic(zoomProgress);
      
      // 相机缩小至单个格子大小
      const targetScale = cellSize / (GRID_SIZE * gridSpacing);
      cameraScale = 1.0 + (targetScale - 1.0) * t;

      // 网格透明度过渡
      currentGridOpacity = Math.max(0, 1.0 - t * 1.5);
      nextGridOpacity = Math.max(0, (t - 0.4) / 0.6);

      if (zoomProgress >= 1.0) {
        // 重启循环
        startCycle(now);
        requestAnimationFrame(draw);
        return;
      }
      break;
  }
}

function startCycle(now: number) {
  activeCells.clear();
  cellBirthTimes.clear();

  activeCells.add(centerCellIdx);
  cellBirthTimes.set(centerCellIdx, now);

  state = 'growing';
  agents = [{ idx: centerCellIdx, prevIdx: centerCellIdx, progress: 0 }];
  lastSimTime = now;
}

无缝循环的秘密：

  • 缩放至单个格子大小
  • 淡出所有格子，仅保留中心格子可见
  • 从中心格子重新开始生长
  • 视觉上形成完美的首尾相接

SVG 栅格化

使用离屏 Canvas 将 SVG Path 转换为 19×19 网格：

function rasterizeSVG() {
  const size = 152;
  const offscreenCanvas = document.createElement('canvas');
  offscreenCanvas.width = size;
  offscreenCanvas.height = size;
  const oCtx = offscreenCanvas.getContext('2d')!;

  // 将 SVG Path 渲染到离屏 Canvas
  oCtx.save();
  oCtx.translate(6, 6);
  oCtx.scale(1.12, 1.12);
  oCtx.fillStyle = 'black';
  oCtx.fill(claudePath2D);
  oCtx.restore();

  // 采样每个网格单元
  const imgData = oCtx.getImageData(0, 0, size, size).data;
  validGridPoints = [];
  validGridSet.clear();

  const cellPixelSize = size / GRID_SIZE;

  for (let gy = 0; gy < GRID_SIZE; gy++) {
    for (let gx = 0; gx < GRID_SIZE; gx++) {
      let filledCount = 0;
      
      // 统计网格内不透明像素数
      for (let py = 0; py < cellPixelSize; py++) {
        for (let px = 0; px < cellPixelSize; px++) {
          const x = Math.floor(gx * cellPixelSize + px);
          const y = Math.floor(gy * cellPixelSize + py);
          const alpha = imgData[(y * size + x) * 4 + 3];
          if (alpha > 50) filledCount++;
        }
      }

      // 填充率 > 15% 则标记为有效格子
      if (filledCount > cellPixelSize * cellPixelSize * 0.15) {
        const idx = gy * GRID_SIZE + gx;
        validGridPoints.push({ gx, gy, idx });
        validGridSet.add(idx);
        cellColors.set(idx, PALETTE[Math.floor(Math.random() * PALETTE.length)]);
      }
    }
  }

  // 构建邻接表（四向连接）
  adjacencyMap.clear();
  const dirs = [[1, 0], [-1, 0], [0, 1], [0, -1]];
  for (const cell of validGridPoints) {
    const neighbors: number[] = [];
    for (const [dx, dy] of dirs) {
      const nx = cell.gx + dx;
      const ny = cell.gy + dy;
      if (nx >= 0 && nx < GRID_SIZE && ny >= 0 && ny < GRID_SIZE) {
        const nidx = ny * GRID_SIZE + nx;
        if (validGridSet.has(nidx)) neighbors.push(nidx);
      }
    }
    adjacencyMap.set(cell.idx, neighbors);
  }

  // 找到最接近中心的格子作为起始点
  let minDist = Infinity;
  for (const cell of validGridPoints) {
    const dist = Math.hypot(cell.gx - 9, cell.gy - 9);
    if (dist < minDist) {
      minDist = dist;
      centerCellIdx = cell.idx;
    }
  }
}

采样策略：

  • 每个网格 8×8 像素
  • 填充率 > 15% → 有效格子
  • 结果：约 170 个有效格子

性能分析

在 60 FPS 下，每帧需要：

170 格子 × (beginPath + roundRect + fill) ≈ 510 次 CPU 调用
+ 5 Agent × (save + rotate + fill + restore) ≈ 20 次调用
+ 2 层网格 × (19×19 圆点) ≈ 722 次调用
= 总计约 1252 次 Canvas API 调用/帧

缓动函数

function easeOutCubic(t: number): number {
  return 1 - Math.pow(1 - t, 3);
}

function easeInOutCubic(t: number): number {
  return t < 0.5
    ? 4 * t * t * t
    : 1 - Math.pow(-2 * t + 2, 3) / 2;
}
  • easeOutCubic：元胞出生动画（快速启动，平滑减速）
  • easeInOutCubic：相机缩放动画（两端平滑）

组件初始化流程

import { onMount, onCleanup } from 'solid-js';

export default function ClaudeLogoAnimation() {
  // ... 变量声明（见上文）

  onMount(() => {
    if (!canvasRef) return;

    // 1. 创建 SVG Path2D 对象
    claudePath2D = new Path2D(CLAUDE_PATH);
    
    // 2. 获取设备像素比（Retina 屏幕 = 2）
    dpr = window.devicePixelRatio || 1;

    // 3. SVG 栅格化 → 生成有效格子和邻接表
    rasterizeSVG();
    
    // 4. 初始化 Canvas 尺寸
    resize();
    
    // 5. 启动第一个循环
    startCycle(performance.now());
    
    // 6. 启动渲染循环
    animationFrameId = requestAnimationFrame(draw);

    // 7. 监听窗口大小变化
    window.addEventListener('resize', resize);
  });

  onCleanup(() => {
    cancelAnimationFrame(animationFrameId);
    window.removeEventListener('resize', resize);
  });

  // Canvas 尺寸调整
  function resize() {
    if (!canvasRef) return;

    // 使用容器尺寸，不是 window 尺寸
    const container = canvasRef.parentElement;
    if (!container) return;

    const rect = container.getBoundingClientRect();
    width = rect.width;
    height = rect.height;
    
    // 设置物理像素尺寸（考虑 DPR）
    canvasRef.width = width * dpr;
    canvasRef.height = height * dpr;
    
    // 应用 DPR 缩放
    const ctx = canvasRef.getContext('2d')!;
    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

    // 重新计算布局
    centerVal = Math.min(width * 0.58, height * 0.58);
    gridSpacing = centerVal / GRID_SIZE;
    cellSize = gridSpacing * 0.88;
    cellRadius = cellSize * 0.25;
  }

  // ... 其他函数
}

// 三次缓出：快速启动，平滑减速
function easeOutCubic(t: number): number {
  return 1 - Math.pow(1 - t, 3);
}

// 三次缓入缓出：两端平滑
function easeInOutCubic(t: number): number {
  return t < 0.5 
    ? 4 * t * t * t 
    : 1 - Math.pow(-2 * t + 2, 3) / 2;
}

// 绘制圆角矩形
function drawRoundedRect(
  ctx: CanvasRenderingContext2D,
  x: number,
  y: number,
  size: number,
  radius: number,
  color: string,
  scale: number
) {
  ctx.save();
  ctx.translate(x + size / 2, y + size / 2);
  ctx.scale(scale, scale);
  ctx.fillStyle = color;
  ctx.beginPath();
  ctx.roundRect(-size / 2, -size / 2, size, size, radius);
  ctx.fill();
  ctx.restore();
}

关键实现细节

坐标系统

  • 网格坐标：(gx, gy) ∈ [0, 18] × [0, 18]
  • 一维索引：idx = gy × 19 + gx
  • Canvas 坐标：中心点 (cx, cy)，格子位置 = cx + (gx - 9) × gridSpacing

时间轴

0ms ~ 填满时间
growing 阶段
BFS 每 150ms 步进，元胞从中心向外扩散
+1000ms
holding 阶段
填满后静止 1 秒
+1600ms → 重启
zooming 阶段
相机缩放至单格子大小，元胞逐渐变回 Logo 形状，循环重启

渲染层级（从后到前）

  1. 背景色 (#FAF9F5)
  2. 未来网格 (zooming 时淡入，透明度 0 → 1)
  3. 当前网格 (始终可见，zooming 时淡出)
  4. 元胞层 (形态变换 + 出生动画)
  5. Agent 层 (旋转的白色 Logo)
  6. 文字层 (HTML 叠加层，CSS z-index: 10)

形态变换阈值

const morphThreshold = 0.7 + zoomProgress * 0.3;
// zoomProgress = 0 → threshold = 0.7
// zoomProgress = 1 → threshold = 1.0

if (scale < morphThreshold) {
  drawClaudeLogoShape(...);  // Logo 形状
} else {
  drawRoundedRect(...);       // 圆角矩形
}

const gx = prevX + (currX - prevX) * agent.progress;
const gy = prevY + (currY - prevY) * agent.progress;
